async function fetchChart(ticker){

  const url=`https://query1.finance.yahoo.com/v8/finance/chart/${ticker}?range=5y&interval=1mo&includePrePost=false`;

  const r=await fetch(url,{headers:{'User-Agent':'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'}});

  if(!r.ok)throw new Error(r.status);

  const d=await r.json();

  const res=d?.chart?.result?.[0];if(!res)throw new Error('no result');

  const meta=res.meta;

  const ts=res.timestamp||[];

  const cl=res.indicators?.adjclose?.[0]?.adjclose||res.indicators?.quote?.[0]?.close||[];

  const price=meta.regularMarketPrice||meta.previousClose;

  const d1pct=meta.regularMarketChangePercent||null;

  const nowMs=Date.now();

  function idxAt(tMs){let b=0,bd=Infinity;ts.forEach((t,i)=>{if(cl[i]==null)return;const diff=Math.abs(t*1000-tMs);if(diff<bd){bd=diff;b=i;}});return b;}

  function pct(f,t){if(!f||!t)return null;return((t-f)/Math.abs(f))*100;}

  const cur=cl[cl.length-1]||price;

  const aum=(meta.totalAssets||0)/1e9||null;

  const isIndian=ticker.endsWith('.NS');

  return{

    price,d1:d1pct,

    w1:pct(cl[idxAt(nowMs-7*86400000)],cur),

    m1:pct(cl[idxAt(nowMs-30*86400000)],cur),

    m3:pct(cl[idxAt(nowMs-91*86400000)],cur),

    m6:pct(cl[idxAt(nowMs-182*86400000)],cur),

    ytd:pct(cl[idxAt(new Date(new Date().getFullYear(),0,2).getTime())],cur),

    y1:pct(cl[idxAt(nowMs-365*86400000)],cur),

    y5:pct(cl[idxAt(nowMs-5*365*86400000)],cur),

    aum,isIndian

  };

}


// ── LINEAR REGRESSION SLOPE across [m1,m3,m6,y1]

// x = [1,3,6,12] (months), y = return values

// returns slope in %/month — positive = accelerating momentum

function computeSlope(d){

  const pts=[[1,d.m1],[3,d.m3],[6,d.m6],[12,d.y1]].filter(p=>p[1]!=null);

  if(pts.length<3)return null;

  const n=pts.length;

  const sx=pts.reduce((s,p)=>s+p[0],0);

  const sy=pts.reduce((s,p)=>s+p[1],0);

  const sxx=pts.reduce((s,p)=>s+p[0]*p[0],0);

  const sxy=pts.reduce((s,p)=>s+p[0]*p[1],0);

  const slope=(n*sxy-sx*sy)/(n*sxx-sx*sx);

  return slope;

}

function retColor(v){if(v==null)return'var(--muted)';return v>=0?'var(--green)':'var(--red)';}


function fmtPrice(ticker){

  const d=live[ticker];if(!d?.price)return'—';

  if(d.isIndian){return currency==='INR'?'₹'+d.price.toFixed(2):'$'+(d.price/fx).toFixed(2);}

function fmtAUM(item){

  const b=live[item.ticker]?.aum||item.aum;if(!b)return'—';

  if(item.ticker?.endsWith('.NS')){return currency==='INR'?'₹'+(b*fx/100).toFixed(0)+' Cr':'$'+b.toFixed(1)+'B';}

  if(currency==='INR'){const cr=b*fx/100;return cr>=10000?'₹'+(cr/100).toFixed(0)+'L Cr':'₹'+Math.round(cr)+' Cr';}

  return b>=100?'$'+b.toFixed(0)+'B':'$'+b.toFixed(1)+'B';

}
